  <VideoTimeAt startTime={285} endTime={306}>

  <VideoTimeAt startTime={309} endTime={325}>

  <VideoTimeAt startTime={331} endTime={353}>

  <VideoTimeAt startTime={355} endTime={369}>

  </VideoTimeAt>



  <VideoTimeAt startTime={373} endTime={394}>

  </VideoTimeAt>



  <VideoTimeAt startTime={395} endTime={416}>
الأدلة كثيرة جداً، ما الدليل على ذلك؟ الأدلة كثيرة جداً، منها هو أساس قضية نزول القرآن مفرقاً. يعني القرآن تعرفون الكفار قالوا لولا نزل عليه القرآن إيش؟ جملة واحدة. كذلك لنثبت به فؤادك، هذا الآن نثبت في فؤادك الواقع الخاص ولا العام؟

  <VideoTimeAt startTime={420} endTime={433}>

  </VideoTimeAt>



  <VideoTimeAt startTime={437} endTime={454}>

  <VideoTimeAt startTime={494} endTime={512}>

  <VideoTimeAt startTime={556} endTime={575}>

  <VideoTimeAt startTime={579} endTime={596}>
هذا كان بسبب سرية معينة، اللي هي سرية عبد الله بن جحش التي أرسلها النبي صلى الله عليه وسلم وكانت في في رجب في بداية رجب، وحصل فيها قتل في الشهر الحرام إلى آخره. وعندك بعد ذلك سورة الأحزاب نزلت مفصلة على أحداث ووقائع معركة الخندق.

  <VideoTimeAt startTime={616} endTime={632}>
